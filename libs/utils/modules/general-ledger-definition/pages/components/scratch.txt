export function groupByType(nodes: IGeneralLedgerDefinition[]) {
    const groups: Record<TGeneralLedgerType, IGeneralLedgerDefinition[]> = {
        Assets: [],
        Liabilities: [],
        Equity: [],
        Revenue: [],
        Expenses: [],
    }

    nodes.forEach((node) => {
        groups[node.general_ledger_type].push(node)
    })

    return groups
}

function AccountItem({ account }: { account: IAccount }) {
    const { setNodeRef, listeners, attributes, transform } = useSortable({
        id: `account-${account.id}`,
        data: {
            type: 'account',
            account,
        },
    })

    const style = {
        transform: CSS.Transform.toString(transform),
    }

    return (
        <div
            ref={setNodeRef}
            style={style}
            {...attributes}
            className="p-2 border rounded bg-muted flex justify-between"
        >
            <span {...listeners} className="cursor-grab">
                {account.name}
            </span>
        </div>
    )
}

function GLNodeCard({ node }: { node: IGeneralLedgerDefinition }) {
    const { setNodeRef, listeners, attributes, transform } = useSortable({
        id: `gl-${node.id}`,
        data: {
            type: 'gl',
            node,
        },
    })

    const style = {
        transform: CSS.Transform.toString(transform),
    }

    return (
        <div
            ref={setNodeRef}
            style={style}
            {...attributes}
            className="bg-card rounded-lg p-3 shadow mb-2"
        >
            <div {...listeners} className="font-semibold cursor-grab">
                {node.name}
            </div>

            {node.accounts && node.accounts.length > 0 && (
                <div className="pl-4 mt-2">
                    <SortableContext
                        items={node.accounts.map((a) => `account-${a.id}`)}
                        // strategy={verticalListSortingStrategy}
                    >
                        {node.accounts.slice(0,1).map((a) => (
                            <AccountItem account={a} key={a.id} />
                        ))}
                    </SortableContext>
                </div>
            )}

            {node.general_ledger_definition_entries &&
                node.general_ledger_definition_entries.length > 0 && (
                    <div className="pl-4 mt-2">
                        <SortableContext
                            items={node.general_ledger_definition_entries.map(
                                (c) => `gl-${c.id}`
                            )}
                            // strategy={verticalListSortingStrategy}
                        >
                            {node.general_ledger_definition_entries.map(
                                (child) => (
                                    <GLNodeCard key={child.id} node={child} />
                                )
                            )}
                        </SortableContext>
                    </div>
                )}
        </div>
    )
}

function GLTypeColumn({
    title,
    nodes,
}: {
    title: string
    nodes: IGeneralLedgerDefinition[]
}) {
    return (
        <div className="bg-secondary/20 rounded-lg p-3 min-h-[400px]">
            <h2 className="font-semibold mb-2">{title}</h2>

            <SortableContext
                items={nodes.map((n) => `gl-${n.id}`)}
                // strategy={verticalListSortingStrategy}
            >
                {nodes.map((node) => (
                    <GLNodeCard key={node.id} node={node} />
                ))}
            </SortableContext>
        </div>
    )
}
function updateNodeTree(
    prev: IGeneralLedgerDefinition[],
    activeId: string,
    overId: string,
    isAccount: boolean
) {
    const clone = structuredClone(prev)
    let movedItem: any = null
    let targetType: TGeneralLedgerType | undefined

    // STEP 1: Deep Search to REMOVE the item and FIND the target's Type
    const findAndRemove = (nodes: IGeneralLedgerDefinition[]) => {
        for (let i = 0; i < nodes.length; i++) {
            const node = nodes[i]

            // If this is our TARGET, grab the type for the moved item
            if (`gl-${node.id}` === overId) {
                targetType = node.general_ledger_type
            }

            // If this is our ACTIVE item (GL Node), remove it
            if (!isAccount && `gl-${node.id}` === activeId) {
                movedItem = nodes.splice(i, 1)[0]
                i-- // Adjust index after splice
            }

            // If we are looking for an ACCOUNT inside this node
            if (isAccount && node.accounts) {
                const accIdx = node.accounts.findIndex(
                    (a) => `account-${a.id}` === activeId
                )
                if (accIdx !== -1) {
                    movedItem = node.accounts.splice(accIdx, 1)[0]
                }
            }

            // Recurse
            if (node.general_ledger_definition_entries) {
                findAndRemove(node.general_ledger_definition_entries)
            }
        }
    }

    findAndRemove(clone)

    // STEP 2: Deep Search to INSERT into the target node
    if (movedItem && targetType) {
        const insert = (nodes: IGeneralLedgerDefinition[]) => {
            for (const node of nodes) {
                if (`gl-${node.id}` === overId) {
                    if (isAccount) {
                        node.accounts = [...(node.accounts ?? []), movedItem]
                    } else {
                        // Update the type of the GL and all its nested children
                        const updateTypeRecursive = (
                            n: IGeneralLedgerDefinition
                        ) => {
                            n.general_ledger_type = targetType!
                            n.general_ledger_definition_entries?.forEach(
                                updateTypeRecursive
                            )
                        }
                        updateTypeRecursive(movedItem)

                        node.general_ledger_definition_entries = [
                            ...(node.general_ledger_definition_entries ?? []),
                            movedItem,
                        ]
                    }
                    return true
                }
                if (
                    node.general_ledger_definition_entries &&
                    insert(node.general_ledger_definition_entries)
                ) {
                    return true
                }
            }
            return false
        }

        const wasInserted = insert(clone)

        // If not inserted into a node, it might be dropped onto a Column (Top Level)
        // You would need logic here to handle dropping directly into GLTypeColumn
    }

    return clone
}
export default function GLTreeView({
    data,
}: {
    data: IGeneralLedgerDefinition[]
}) {
    const [tree, setTree] = useState<IGeneralLedgerDefinition[]>(data)

    useEffect(() => {
        if (!data) return
        setTree(data)
    }, [data])

    const sensors = useSensors(
        useSensor(PointerSensor, { activationConstraint: { distance: 8 } })
    )

    function moveAccount(activeId: string, overId: string) {
        const accountId = activeId.replace('account-', '')
        const targetGlId = overId.replace('gl-', '')

        setTree((prev) => {
            const clone = structuredClone(prev)
            let accountToMove: IAccount | null = null
            let targetType: TGeneralLedgerType | null = null

            // Phase 1: Find account and the TARGET node's type
            function findTargetAndRemove(nodes: IGeneralLedgerDefinition[]) {
                for (const node of nodes) {
                    if (node.id === targetGlId) {
                        targetType = node.general_ledger_type // Capture the type of the destination
                    }
                    if (node.accounts) {
                        const idx = node.accounts.findIndex(
                            (a) => a.id === accountId
                        )
                        if (idx !== -1) {
                            accountToMove = node.accounts.splice(idx, 1)[0]
                        }
                    }
                    if (node.general_ledger_definition_entries) {
                        findTargetAndRemove(
                            node.general_ledger_definition_entries
                        )
                    }
                }
            }

            findTargetAndRemove(clone)

            // Phase 2: Insert and Update Type
            if (accountToMove && targetGlId) {
                function insert(nodes: IGeneralLedgerDefinition[]) {
                    for (const node of nodes) {
                        if (node.id === targetGlId) {
                            // Crucial: Update the account's parent reference if your API needs it
                            node.accounts = [
                                ...(node.accounts ?? []),
                                accountToMove!,
                            ]
                            return true
                        }
                        if (node.general_ledger_definition_entries?.length) {
                            if (insert(node.general_ledger_definition_entries))
                                return true
                        }
                    }
                    return false
                }
                insert(clone)
            }

            return clone
        })
    }
    function moveGL(activeId: string, overId: string) {
        const activeGLId = activeId.replace('gl-', '')
        const overGLId = overId.replace('gl-', '')

        setTree((prev) => {
            const clone = structuredClone(prev)

            const findNode = (
                nodes: IGeneralLedgerDefinition[]
            ): IGeneralLedgerDefinition | null => {
                for (const node of nodes) {
                    if (node.id === activeGLId) return node

                    if (node.general_ledger_definition_entries) {
                        const found = findNode(
                            node.general_ledger_definition_entries
                        )
                        if (found) return found
                    }
                }

                return null
            }

            const findParentArray = (
                nodes: IGeneralLedgerDefinition[]
            ): IGeneralLedgerDefinition[] | null => {
                if (nodes.some((n) => n.id === activeGLId)) return nodes

                for (const node of nodes) {
                    if (node.general_ledger_definition_entries) {
                        const found = findParentArray(
                            node.general_ledger_definition_entries
                        )

                        if (found) return found
                    }
                }

                return null
            }

            const parentArray = findParentArray(clone)

            if (!parentArray) return prev

            const oldIndex = parentArray.findIndex((n) => n.id === activeGLId)
            const newIndex = parentArray.findIndex((n) => n.id === overGLId)

            if (oldIndex === -1 || newIndex === -1) return prev

            const moved = arrayMove(parentArray, oldIndex, newIndex)

            moved.forEach((node, i) => {
                node.index = i
            })

            return clone
        })
    }

    function handleDragEnd(event: DragEndEvent) {
        const { active, over } = event
        if (!over || active.id === over.id) return

        const activeType = active.data.current?.type // 'gl' or 'account'
        const overType = over.data.current?.type // 'gl'

        // We only allow dropping onto GL nodes in this logic
        if (overType === 'gl') {
            setTree((prev) =>
                updateNodeTree(
                    prev,
                    active.id.toString(),
                    over.id.toString(),
                    activeType === 'account'
                )
            )
        }
    }

    const grouped = groupByType(tree)

    return (
        <DndContext
            collisionDetection={closestCorners}
            onDragEnd={handleDragEnd}
            sensors={sensors}
        >
            <div className="grid grid-cols-4">
                {GENERAL_LEDGER_TYPE.map((type) => (
                    <GLTypeColumn
                        key={type}
                        nodes={grouped[type]}
                        title={type}
                    />
                ))}
            </div>
        </DndContext>
    )
}
